Features

Three primitives. No mystery.

Events

Ingest structured events. Get ids back. Fan-out later.

LANG: JS 
import { Gridforge } from "@gridforge/sdk";

const gf = new Gridforge({ token: process.env.TOKEN });

const res = await gf.events.ingest({
  event: "build.complete",
  meta: { sha: "a1b2c3" }
});

console.log(res.id);
import os
from gridforge import Gridforge

gf = Gridforge(token=os.environ["TOKEN"])

res = gf.events.ingest(
  event="build.complete",
  meta={"sha": "a1b2c3"},
)

print(res["id"])
curl -X POST "https://api.gridforge.dev/v1/events" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"event":"build.complete","meta":{"sha":"a1b2c3"}}'

Auth

Mint short-lived tokens. Rotate keys. Audit everything.

LANG: JS 
import { Gridforge } from "@gridforge/sdk";

const gf = new Gridforge({ token: process.env.TOKEN });

const token = await gf.auth.mint({
  subject: "user_42",
  ttlSeconds: 900,
  scopes: ["events:write"]
});

console.log(token.value);
import os
from gridforge import Gridforge

gf = Gridforge(token=os.environ["TOKEN"])

token = gf.auth.mint(
  subject="user_42",
  ttl_seconds=900,
  scopes=["events:write"],
)

print(token["value"])
curl -X POST "https://api.gridforge.dev/v1/auth/mint" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"subject":"user_42","ttlSeconds":900,"scopes":["events:write"]}'

Edge KV

Read/write small values near users. Explicit limits. Explicit consistency.

LANG: JS 
import { Gridforge } from "@gridforge/sdk";

const gf = new Gridforge({ token: process.env.TOKEN });

await gf.kv.put("flags:checkout", "on", { ttlSeconds: 60 });

const v = await gf.kv.get("flags:checkout");
console.log(v);
import os
from gridforge import Gridforge

gf = Gridforge(token=os.environ["TOKEN"])

gf.kv.put("flags:checkout", "on", ttl_seconds=60)

v = gf.kv.get("flags:checkout")
print(v)
curl -X PUT "https://api.gridforge.dev/v1/kv/flags:checkout" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"value":"on","ttlSeconds":60}'
